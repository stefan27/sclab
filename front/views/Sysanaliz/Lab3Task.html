            INITIAL 	X$Facval,100
            GENERATE	1
            SAVEVALUE	Answer,(Factorial(X$Facval))
            TERMINATE	1


            PROCEDURE Factorial(Arg1) BEGIN
            TEMPORARY Result;

            Result=1;

            WHILE(Arg1>1) DO BEGIN
            Result=Result#Arg1;
            Arg1=(Arg1-1);
            END;

            RETURN Result;

            END;
